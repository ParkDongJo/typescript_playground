# 타입추론

아래의 코드를 분석하는 걸로!!

```typescript
/**
 * useQueries의 결과값을 각 api의 return 값의 타입 추론을 위한 커스텀 useQueries
 * 주의 : useQuery의 option 속성의 경우 타입이추론 되지 않습니다.
 * 함수의 경우 매개변수의 타입을 잡기 위해 type단언문을 사용 해야 할 수 있습니다.
 * @param queries
 * @returns useQueries
 */
export function useQueriesTyped<TQueries extends readonly UseQueryOptions[]>(
  queries: [...TQueries]
): {
  [ArrayElement in keyof TQueries]: UseQueryResult<
    TQueries[ArrayElement] extends { select: infer TSelect }
      ? TSelect extends (data: any) => any
        ? ReturnType<TSelect>
        : never
      : Awaited<
          ReturnType<
            NonNullable<
              Extract<
                TQueries[ArrayElement],
                UseQueryOptions
              >["queryFn"] extends (...args: any) => any
                ? Extract<TQueries[ArrayElement], UseQueryOptions>["queryFn"]
                : never
            >
          >
        >
  >;
} {
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  return useQueries({
    queries,
  }) as any;
}
```
